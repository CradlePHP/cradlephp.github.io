Cradle PHP

An unopinionated PHP7 micro framework built on traits.

1. Epic Routing


cradle()->get('/foo/:bar', function($request, $response) {
    $response->setContent('Hello, World');
});

cradle()->post('/foo/*(/:start){0,1}', function($request, $response) {
    $response->setContent('Hello, World');
});

2. Limitless Events


cradle()->on('create-user', function(...$data) {
    //insert to database
});

cradle()->on('Show a %s message', function() {
    $meta = $this->getEventHandler()->getMeta();
    echo $meta['variables'][0];
});

cradle()->on('#^foo$#', function() {
    echo 'bar';
});

3. Granular Middleware


cradle()->preprocess(function($request, $response) {
    session_start();
});

cradle()->postprocess(function($request, $response) {
    session_write_close();
});

cradle()->error(function($request, $response, $error) {
    $response->setContent('Something went wrong');
});

cradle()->all(
    '/foo/bar',
    'must-be-logged-in',
    function($request, $response) {
        //and an admin
    },
    function($request, $response) {
        $response->setContent('Hello, World');
    },
    'Controller@addView'
);

4. Use Any Template Engine


use Philo\Blade\Blade;
use Cradle\Handlebars\HandlebarsHandler;

//blade
cradle()->get('/foo/:bar', function($request, $response) {
    $blade = new Blade('/path/to/views', '/path/to/cache');
    $content = $blade->view()->make('hello')->render();
    $response->setContent($content);
});

//twig
cradle()->get('/foo/:bar', function($request, $response) {
    $loader = new Twig_Loader_Array(['index' => 'Hello {{ name }}!']);
    $twig = new Twig_Environment($loader);
    $content = $twig->render('index', $request->getStage());
    $response->setContent($content);
});

//handlebars
cradle()->get('/foo/:bar', function($request, $response) {
    $template = HandlebarsHandler::i()->compile('/path/to/template');
    $content = $template($request->getStage());
    $response->setContent($content);
});
