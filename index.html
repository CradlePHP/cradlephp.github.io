Cradle PHP

in 6 reasons.

1. It's Pseudo Coding Powered by English

<?php
cradle()->get(
    '/foo/bar', 
    'Get Foo Bar from database',
    array(
        'Foo Bar data Found',
        'Render Foo Bar template',
        'Render with Custom Page'
    ),
    array(
        'Foo Bar data Not Found',
        'Flash error',
        'Redirect to homepage',
    )
);

2. Where You Can Use Any Controller

<?php
cradle()->get(
    '/foo/bar', 
    'Controller@method',
    'Controller::method'
);

3. And Use Any Framework

<?php
/**
 * Hello Laravel !
 */
$app->get('/', cradle()->export('Get Foo Bar from database'));

4. And Easily Troubleshoot

<?php
cradle()->post(
    '/signup', 
    'Validate sign up',
    array(
    	'All Good ?',
        'Insert to Database',
        'debug://WTH Am I here ?'
        //'Send an email',
        function($request, $response) {
       		//Try sending mail manually ...
        }
        'Render Login',
        'Render Custom Page'
    )
);

5. And Unopinionated

<?php
cradle()
	->post('/signup', 'Validate sign up')
    ->flow(
    	'Validate sign up',
        'Prepare Data',
        include('path/to/callback.php')
    );

...

//path/to/callback.php
return function($request, $response) {
	//look Ma no classes!
	$response->setContent('valid');
    //fork a process
    $this->triggerEvent('All Good ?', $request, $response);
};

6. And Redistribute as a Composer Package

<?php

cradle()
	->register('cblanquera/cradle-auth')
    ->post(
        '/auth/create', 
        Flow::auth()->create->prepare,
        Flow::auth()->create->valid,
        array(
        	Flow::auth()->create->yes,
        	Flow::auth()->create->task,
            'Flash success',
            'Render with Custom Page'
        )
    );
